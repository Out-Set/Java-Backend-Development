-- Create a table to store JSON data
CREATE TABLE my_json_table (
    id NUMBER GENERATED BY DEFAULT ON NULL AS IDENTITY,
    json_data CLOB
);

-- Insert JSON data using JSON_OBJECT
INSERT INTO my_json_table (json_data)
VALUES (JSON_OBJECT('key1' VALUE 'value1', 'key2' VALUE 'value2'));

-- Alternatively, you can use a JSON string directly
INSERT INTO my_json_table (json_data)
VALUES ('{"name": "sawan", "add": "noida sector 63, up"}');

-- Query to select the inserted data
SELECT * FROM my_json_table;
